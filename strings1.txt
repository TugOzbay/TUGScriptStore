my $str = "The red and white boys of Arsenal";

    print "$word\n";
}

print " \n\n\n";
print "All words: @words\n";		# print the whole array @words
print "Second word: $words[1]\n";	# print the slot [1] which is the second word in the @word array
# print "Second word: @words[1]\n";	# print the slot [1] which is the second word in the @word array

exit 0;
